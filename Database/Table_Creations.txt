
//Test table
CREATE TABLE `Owlhowl`.`test` (
  `testID` INT NOT NULL AUTO_INCREMENT,
  `message` VARCHAR(45) NULL,
  `lat` VARCHAR(45) NULL,
  `lng` VARCHAR(45) NULL,
  PRIMARY KEY (`testID`))
COMMENT = 'Test table for the app to test connection to.';

//Device Table
CREATE TABLE `device` (
  `deviceID` int(11) NOT NULL AUTO_INCREMENT,
  `name` varchar(45) NOT NULL,
  PRIMARY KEY (`deviceID`,`name`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1
COMMENT='This table is to hold the unique identification for each device.';

//Messages table
CREATE TABLE `Owlhowl`.`message` (
  `messageID` INT NOT NULL AUTO_INCREMENT,
  `message` VARCHAR(140) NOT NULL,
  `lat` DOUBLE NOT NULL,
  `lng` DOUBLE NOT NULL,
  `deviceID` INT NOT NULL COMMENT 'FK To connect back to a posting device.',
  `handleID` VARCHAR(45) NOT NULL COMMENT 'Relate a given handle to a message.',
  PRIMARY KEY (`messageID`))
COMMENT = 'This table is to hold the messages and the location in which they were posted.';

//Handle Table
CREATE TABLE `Owlhowl`.`handle` (
  `handleID` INT NOT NULL AUTO_INCREMENT,
  `handle` VARCHAR(45) NOT NULL,
  `deviceID` VARCHAR(45) NOT NULL COMMENT 'FK to relate a given handle to a single device.',
  PRIMARY KEY (`handleID`))
COMMENT = 'This table will be truncated daily.';

//Comments table (if we get that far)
CREATE TABLE `Owlhowl`.`comment` (
  `commentID` INT NOT NULL AUTO_INCREMENT,
  `comment` VARCHAR(140) NOT NULL,
  `messageID` INT NOT NULL COMMENT 'FK to relate it to the message it was posted to.',
  PRIMARY KEY (`commentID`))
COMMENT = 'This table is for a comments to a post';
